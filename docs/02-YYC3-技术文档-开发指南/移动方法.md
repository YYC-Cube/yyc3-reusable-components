**方案2: 动态检测**

```typescript
// utils/pathResolver.ts
export function resolveWorkspacePath(): string {
  if (process.env.YYC3_WORKSPACE) {
    return process.env.YYC3_WORKSPACE;
  }

  // 检测常见的工作空间位置
  const possiblePaths = [
    './workspace',
    '../workspace',
    './projects',
    './YYC3-AI-Code',
  ];

  for (const path of possiblePaths) {
    if (fs.existsSync(path)) {
      return path;
    }
  }

  return './workspace'; // 默认值
}
```
